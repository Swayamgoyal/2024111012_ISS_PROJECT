Text Analysis Tool

Try my interactive text analyzer that can count letters, words, pronouns, prepositions, and articles.

function analyzeText(text) {
  const words = text.split(/\s+/);
  const letters = text.match(/[a-zA-Z]/g);
  const pronouns = words.filter(w => 
    ['I', 'you', 'he', 'she', 'we', 'they'].includes(w));
    
  return {
    words: words.length,
    letters: letters.length,
    pronouns: pronouns.length
  };
}
